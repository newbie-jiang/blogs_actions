main.c

#include <stdio.h>

int main(void) {
    printf("Hello, CMake!\n");
    return 0;
}

CMakeLists.txt

cmake_minimum_required(VERSION 3.5)
project(TestCMake C)
add_executable(test_cmake main.c)

编译

cmake -S . -B build -G "MinGW Makefiles" 
cmake --build build
cd build
.\test_cmake.exe
